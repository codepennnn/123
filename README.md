SELECT RECEIPT_NO,
       REGEXP_SUBSTR(RECEIPT_NO, '\d+$') AS EXTRACTED_NUMBER
FROM CIDRDB.T_BOOKING_DETAILS
WHERE TO_NUMBER(REGEXP_SUBSTR(RECEIPT_NO, '\d+$')) = (
    SELECT MAX(TO_NUMBER(REGEXP_SUBSTR(RECEIPT_NO, '\d+$')))
    FROM CIDRDB.T_BOOKING_DETAILS
    WHERE REGEXP_LIKE(RECEIPT_NO, '\d+$')
);
